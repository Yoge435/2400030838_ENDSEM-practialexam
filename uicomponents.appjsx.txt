// README.md
# Reusable UI Component Library

This repository demonstrates a small reusable UI component library (Button, Input, Card) built with React + TypeScript.

Features:
- Button, Input, Card components that accept props for customization
- Two example pages (Home, Dashboard) using the components with different props
- Easy to extract into a component package

Run locally:
bash
npm install
npm run dev

// src/components/Button.tsx
import React from 'react'

export type ButtonProps = React.ButtonHTMLAttributes<HTMLButtonElement> & {
  variant?: 'primary' | 'ghost' | 'danger'
  size?: 'sm' | 'md' | 'lg'
  children?: React.ReactNode
}

export default function Button({ variant = 'primary', size = 'md', children, className = '', ...rest }: ButtonProps){
  const sizeClass = size === 'sm' ? 'padding:6px 10px' : size === 'lg' ? 'padding:12px 16px' : ''
  const base = `btn ${variant === 'primary' ? 'btn-primary' : 'btn-ghost'} `
  return (
    <button className={base + className} {...rest}>
      {children}
    </button>
  )
}

// src/components/Input.tsx
import React from 'react'

export type InputProps = React.InputHTMLAttributes<HTMLInputElement> & {
  label?: string
  error?: string
}

export default function Input({ label, error, ...rest }: InputProps){
  return (
    <label style={{ display: 'block', gap:8 }}>
      {label && <div style={{ marginBottom:6, color:'#374151', fontWeight:600 }}>{label}</div>}
      <input className="input" {...rest} />
      {error && <div style={{ color: '#ef4444', marginTop:6 }}>{error}</div>}
    </label>
  )
}

// src/components/Card.tsx
import React from 'react'

export type CardProps = React.HTMLAttributes<HTMLDivElement> & {
  title?: React.ReactNode
  footer?: React.ReactNode
}

export default function Card({ title, footer, children, className = '', ...rest }: CardProps){
  return (
    <div className={"card " + className} {...rest}>
      {title && <h3 style={{ marginTop:0, marginBottom:8 }}>{title}</h3>}
      <div>{children}</div>
      {footer && <div style={{ marginTop:12 }}>{footer}</div>}
    </div>
  )
}

// src/App.tsx
import React from 'react'
import Home from './pages/Home'
import Dashboard from './pages/Dashboard'

export default function App(){
  const [page, setPage] = React.useState<'home'|'dashboard'>('home')
  return (
    <div>
      <nav style={{ padding:16, display:'flex', gap:8 }}>
        <button onClick={() => setPage('home')}>Home</button>
        <button onClick={() => setPage('dashboard')}>Dashboard</button>
      </nav>
      {page === 'home' ? <Home /> : <Dashboard />}
    </div>
  )
}
